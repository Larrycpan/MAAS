2.CNV calling

Here we use the R package epiAneufinder to call copy numbers for each cell.

# Clear workspace and load libraries
rm(list = ls())
library(epiAneufinder)
epiAneufinder(input="./CRC/CRC-3-11773/raw_data/CRC-3-11773/outs/fragments.tsv.gz",
              outdir="./",
              blacklist="./hg38-blacklist.v2.bed",
              windowSize = 1e5,
              genome = "BSgenome.Hsapiens.UCSC.hg38",
              exclude = c('chrX','chrY','chrM'),
              reuse.existing = TRUE,
              title_karyo="",
              ncores = 60,
              minFrags = 15000)

After finishing running epiAneufinder, you will get the a folder named epiAneufinder_results in the current directory. Then we should perform CNV visualization and clustering to identify aneuploid tumor cells.

library(data.table)
library(dplyr)
library(ComplexHeatmap)
library(circlize)

#Obtain the range of chromosomes splitted to short (p) and long arms (q)
chrom.range <- fread("http://hgdownload.cse.ucsc.edu/goldenpath/hg38/database/cytoBand.txt.gz",
           col.names = c("chrom","Start","End","arm","gieStain"))
chrom.list <- paste0("chr",c(1:22, "X"))
chrom.range <- chrom.range[which(chrom.range$chrom %in% chrom.list),]
chrom.range <- dplyr::select(chrom.range, -c(gieStain))
chrom.range$arm <- substr(chrom.range$arm, 1,1)
chrom.range <- chrom.range %>%
  mutate(chrom_band = paste(chrom, arm, sep = ""))
chrom.range <- chrom.range %>%
  group_by(chrom_band) %>%
  dplyr::summarise(start = min(Start),
            end = max(End))
chrom.range$chrom <- gsub("([[:alnum:]]+)[[:alpha:]]+$", "\\1", chrom.range$chrom_band)

Then we manipulate the epiAneufinder results to obatin a cell-by-CNV matrix.

#  One can run the following code directly.
df <- read.table("./epiAneufinder_results/results_table.tsv", row.names = 1)
input.range <- df[,1:3]
df$range <- paste0(df$seq, ":", df$start, "-", df$end)
df <- dplyr::select(df, -c(seq, start, end))
rownames(df) <- df$range
df <- dplyr::select(df, -c(range))
df <- as.data.frame(t(df))
deter.arm <- c()
for(i in 1:nrow(input.range)){
  chr.tmp <- input.range$seq[i]
  start.tmp <- input.range$start[i]
  end.tmp <- input.range$end[i]
  chrom.range.tmp <- chrom.range[which(chrom.range$chrom == chr.tmp &
                                         chrom.range$start <= start.tmp &
                                         chrom.range$end >= end.tmp),]
  deter.arm[i] <- chrom.range.tmp$chrom_band
}
deter.arm <- substr(deter.arm, nchar(deter.arm), nchar(deter.arm))
colnames(df) <- paste0(colnames(df), "(", deter.arm, ")")
columns <- colnames(df)

To identify tumor cells, we perform K-means clustering to distinguish aneuploidy from normal cells that have low frequency of CNVs and low-degree of clonal copy numbers.

# K-means clustering
set.seed(2)
clu <- kmeans(df, centers = 3)$cluster
clu <- as.data.frame(clu)
clu <- clu[order(clu$clu),,drop = F]
df <- df[rownames(clu),]

# Visualization
chr_full_name <- gsub("chr([0-9XYM]+):([0-9]+)-([0-9]+)\\(([pq])\\).*", "chr\\1\\4", colnames(df))
ann.col <- c(rep(c("black", "#E0E0E0"), ceiling(length(unique(chr_full_name)) / 2)))
names(ann.col) <- unique(chr_full_name)
ann.row <- c("#D95F02", "#1B9E77", "#B07164")
names(ann.row) <- c("1", "2", "3")


heat.col <- colorRamp2(c(0, 1, 2), c("#408CBF", "white", "#D15848"))
pdf("totalcell.cnv.heatmap.pdf", width = 9, height = 4.5)
ha <- HeatmapAnnotation(chr = chr_full_name,
                        col = list(chr = ann.col),
                        show_annotation_name = F,
                        show_legend = F)
row_ha <- rowAnnotation(Type = clu$clu,
                        col = list(Type = ann.row),
                        show_annotation_name = F,
                        show_legend = F)
Heatmap(as.matrix(df), name="CNV",
        top_annotation = ha,
        left_annotation = row_ha,
        col = heat.col,
        column_names_gp = gpar(fontsize=10),
        cluster_rows = F,
        cluster_columns = F,
        show_row_dend = F,
        show_row_names = F,
        show_column_names = F)
dev.off()

According to the heatmap, we can see that cluster 1 and cluster 3 exhibit aneuploidy and clonal pattern of copy numbers, while cluster 2 demonstrate low-degree of CNVs. Hence, we annotate cells in cluster 1 and cluster 3 as tumor cells.
