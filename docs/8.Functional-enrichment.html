8.Functional enrichment

We can finally analyze the biological function enrichment between the subpopulations based on the cancer hallmark signatures, which can be downloaded from MSigDB.

# Clear working space and load libraries
rm(list = ls())
library(limma)
library(GSEABase)
library(GSVA)
library(pheatmap)
clusterFile <- "../6.Cluster/maas.pca.rds"
gmtFile <- "h.all.v2023.1.Hs.symbols.gmt"
# Load gene activity score
data <- readRDS("../1.Peak_calling/geneScoreMatrix.rds")
colnames(data) <- gsub("CRC#", "", colnames(data))
cluster <- readRDS(clusterFile)
data <- data[,rownames(cluster)]
# Single cell GSVA
geneSets <- getGmt(gmtFile, geneIdType=SymbolIdentifier())
gsvaResult <- gsva(as.matrix(data), geneSets, min.sz = 10, verbose = T, parallel.sz = 50)
# gsvaOut <- rbind(id = colnames(gsvaResult), gsvaResult)
# write.table(gsvaOut, file="gsvaOut.txt", sep="\t", quote=F, col.names=F)

gsvaResult <- as.data.frame(t(gsvaResult))
colnames(gsvaResult) <- gsub("HALLMARK_", "", colnames(gsvaResult))
sameSample <- intersect(row.names(gsvaResult), row.names(cluster))
gsvaResult1 <- gsvaResult[sameSample,,drop=F]
cluster <- cluster[sameSample,,drop=F]
gsvaCluster <- cbind(gsvaResult1, Subpopulation = cluster$Subpopulation)
# Differential analysis
adj.P.Val.Filter=0.05
allType=as.vector(gsvaCluster$Subpopulation)
comp=combn(levels(factor(allType)), 2)
for(i in 1:ncol(comp)){
  #样品分组
  treat=gsvaCluster[gsvaCluster$Subpopulation==comp[2,i],]
  con=gsvaCluster[gsvaCluster$Subpopulation==comp[1,i],]
  data=rbind(con, treat)
  #差异分析
  Type=as.vector(data$Subpopulation)
  ann=data.frame(data[,ncol(data)])
  row.names(ann)=row.names(data)
  names(ann)="Subpopulation"
  data=t(data[,-(ncol(data))])
  design=model.matrix(~0+factor(Type))
  colnames(design)=levels(factor(Type))
  fit=lmFit(data, design)
  contrast=paste0(comp[2,i], "-", comp[1,i])
  cont.matrix=makeContrasts(contrast, levels=design)
  fit2=contrasts.fit(fit, cont.matrix)
  fit2=eBayes(fit2)
  
  #输出所有通路的差异情况
  allDiff=topTable(fit2,adjust='fdr',number=200000)
  allDiffOut=rbind(id=colnames(allDiff),allDiff)
  write.table(allDiffOut, file=paste0(contrast, ".all.txt"), sep="\t", quote=F, col.names=F)
  
  #输出显著的差异
  diffSig=allDiff[with(allDiff, (abs(logFC) > 0.075 & adj.P.Val < adj.P.Val.Filter )), ]
  diffSigOut=rbind(id=colnames(diffSig),diffSig)
  write.table(diffSigOut, file=paste0(contrast, ".diff.txt"), sep="\t", quote=F, col.names=F)
  #聚类颜色
  my_color <- c("#5050FF", "#CE3D32", "#749B58", "#F0E685", "#00BFC4", "#B79F00", "#00BA38")
  heatmap.BlBkRd <- c("#00DDFF","#32ABAA","#125456","#000000","#510000","#A20000","#F30000")
  ann_colors=list()
  CluCol=my_color[1:length(levels(factor(allType)))]
  names(CluCol)=levels(factor(allType))
  ann_colors[["Subpopulation"]]=CluCol[c(comp[1,i], comp[2,i])]

  #绘制差异通路
  termNum=30
  diffTermName=as.vector(rownames(diffSig))
  diffLength=length(diffTermName)
  if(diffLength<termNum){termNum=diffLength}
  hmGene=diffTermName[1:termNum]
  hmExp=data[hmGene,]
  
  p1 <- pheatmap::pheatmap(hmExp,
           annotation=ann,
           annotation_colors = ann_colors,
           color = colorRampPalette(c("#003A7A", "white", "#A20000"))(60),
           cluster_cols =F,
           show_colnames = F,
           # gaps_col=as.vector(cumsum(table(Type))),
           scale="row",
           cluster_row = T,
           fontsize = 10,
           fontsize_row=8,
           fontsize_col=10,
           cellheight = 20,
           treeheight_row = 0)
  ggsave(paste0(contrast,".heatmap_hallmark.pdf"), p1, height = 6, width = 6)
}
