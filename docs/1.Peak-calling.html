1.Peak calling

Here we use the R package ArchR to processing scATAC-seq peak data and call peaks. The example data has been updated to Zenodo

# Clear workspace and load libraries
rm(list = ls())
library(ArchR)
library(dplyr)
library(HGNChelper)

Create an arrow file

First, we generate an arrow file for filtering low-quality cells and peaks.

addArchRThreads(threads = 40)
addArchRGenome("hg38")
inputFiles <- c("./CRC-3-11773/raw_data/CRC-3-11773/outs/fragments.tsv.gz")
names(inputFiles) <- c("CRC")
ArrowFiles <- createArrowFiles(
  inputFiles = inputFiles,
  sampleNames = names(inputFiles),
  minTSS = 3,
  minFrags = 1000, 
  addTileMat = T,
  addGeneScoreMat = T
)

Create ArchR project

For downstream analysis, we create ArchR project to perform dimension reduction and clustering.

projHeme1 <- ArchRProject(ArrowFiles = ArrowFiles, outputDirectory = "CRC", copyArrows = F)
# The parameters can be changed based on user's own data
projHeme2 <- addIterativeLSI(ArchRProj = projHeme1, useMatrix = "TileMatrix", name = "IterativeLSI",
                             iterations = 2, clusterParams = list(resolution = c(0.2), 
                                                                  sampleCells = 10000, 
                                                                  n.start = 10
                                                                  ), 
                             varFeatures = 20000, 
                             dimsToUse = 1:30
                             )
projHeme2 <- addClusters(input = projHeme2, reducedDims = "IterativeLSI", method = "Seurat", 
                         name = "Clusters", resolution = 0.5, force = T)

Peak calling

As suggested, we use MACS2 for peak calling. Users should specify the their own path to MACS2. In the study, we set the window of each peak as 5000bp using the parameter extendSummits = 2500.

projHeme3 <- addGroupCoverages(ArchRProj = projHeme2, groupBy = "Clusters")
projHeme3 <- addReproduciblePeakSet(ArchRProj = projHeme3, groupBy = "Clusters", 
                                    pathToMacs2 = "/usr/local/bin/macs2",
                                    shift = -100,
                                    extsize = 200,
                                    extendSummits = 2500,
                                    force = T)
projHeme4 <- addPeakMatrix(projHeme3, force = T)

Manipulate called peaks

# Save the peak name formatted as "chrXX:XXXXXX-XXXXXX"
peak.Matrix.obj <- getMatrixFromProject(projHeme4, useMatrix = "PeakMatrix")
peak_ranges <- rowRanges(peak.Matrix.obj)
peak_names <- paste0(seqnames(peak_ranges), ":", start(peak_ranges), "-", end(peak_ranges))

# Get peak matrix, cell in columns and peaks in row
peak.matrix <- as.data.frame(assay(getMatrixFromProject(projHeme4, useMatrix = "PeakMatrix")))
rownames(peak.matrix) <- peak_names
saveRDS(peak.matrix, "peak.matrix.rds")

Cell annotation using gene activity scores

Here we use scType to automatically annotate cell types. This step can be used for multiple single-cell data analysis, such as cell-cell communications and pseudo-time trajectory reconstruction etc.

# Obtain gene score matrix
geneMatrix <- getMatrixFromProject(projHeme4, useMatrix = "GeneScoreMatrix")
geneScoreMatrix <- as.data.frame(geneMatrix@assays@data@listData[["GeneScoreMatrix"]])
rownames(geneScoreMatrix) <- geneMatrix@elementMetadata@listData[["name"]]
saveRDS(geneScoreMatrix, "geneScoreMatrix.rds")
# scType analysis
# load gene set preparation function
source("gene_sets_prepare.R")
# load cell type annotation function
source("sctype_score_.R")
# DB file
db_ = "ScTypeDB_full.xlsx"
tissue = "Intestine" 

# prepare gene sets
gs_list <- gene_sets_prepare(db_, tissue)

# get cell-type by cell matrix
es.max <- sctype_score(scRNAseqData = as.matrix(geneScoreMatrix), scaled = TRUE, 
                       gs = gs_list$gs_positive, gs2 = gs_list$gs_negative) 

# merge by cluster
cell.meta <- as.data.frame(getCellColData(projHeme4, select = "Clusters"))
cL_resutls <- do.call("rbind", lapply(unique(cell.meta$Clusters), function(cl){
  es.max.cl = sort(rowSums(es.max[ ,rownames(cell.meta[cell.meta$Clusters==cl,,drop=F])]), decreasing = !0)
  head(data.frame(cluster = cl, type = names(es.max.cl), scores = es.max.cl, ncells = sum(cell.meta$Clusters==cl)), 10)
}))
sctype_scores = cL_resutls %>% group_by(cluster) %>% top_n(n = 1, wt = scores)  

# set low-confident (low ScType score) clusters to "unknown"
sctype_scores$type[as.numeric(as.character(sctype_scores$scores)) < sctype_scores$ncells/4] = "Unknown"

cell.meta$customclassif = ""
for(j in unique(sctype_scores$cluster)){
  cl_type = sctype_scores[sctype_scores$cluster==j,]; 
  cell.meta$customclassif[cell.meta$Clusters == j] = as.character(cl_type$type[1])
}
projHeme5 <- addCellColData(projHeme4, data = cell.meta$customclassif, cells = projHeme4$cellNames, name = "Cell_type")

p <- plotEmbedding(ArchRProj = projHeme5, colorBy = "cellColData", name = "Cell_type", embedding = "UMAP")
plotPDF(p, name = "Plot-UMAP-Cell-type.pdf", ArchRProj = projHeme5, addDOC = FALSE, width = 5, height = 5)

# Save confident non-tumor cells
normal.cell <- rownames(cell.meta[cell.meta$customclassif != "Intestinal epithelial cells",])
normal.cell <- data.frame(V1 = normal.cell, V2 = normal.cell)
 # This file can be used for bam file split according to barcode names
write.table(normal.cell, "normal.cell.txt", sep = '\t', quote = F, row.names = F)
