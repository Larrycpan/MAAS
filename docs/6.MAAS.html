6.MAAS

# Clear working space and load libraries
rm(list = ls())
library(dplyr)
library(Rcpp)
# Load MAAS Rcpp functions
for (i in list.files("./src/", pattern="MAAS*")) {
  sourceCpp(paste0("./src/", i))
}
# Load data
# Load peak profile
peak.raw <- readRDS("./data/peak.matrix.rds")
peak.raw <- as.data.frame(t(peak.raw))
# Remove prefix of barcodes
rownames(peak.raw) <- gsub(".*#(.*)", "\\1", rownames(peak.raw))

# Load CNV profile
cnv.raw <- read.csv("./data/cancer.cnv.profile.csv", row.names = 1)
cnv.raw <- cnv.raw + 1
# Change barcode name
rownames(cnv.raw) <- gsub("^cell\\.", "", rownames(cnv.raw))
rownames(cnv.raw) <- gsub("\\.", "-", rownames(cnv.raw))
# Change bin name
pattern <- "chr(\\w+)\\.(\\d+)\\.(\\d+)\\.(\\w+)\\."
colnames(cnv.raw) <- gsub(pattern, "chr\\1:\\2-\\3", colnames(cnv.raw))
colnames(cnv.raw) <- gsub(pattern, "chr\\1:\\2-\\3", colnames(cnv.raw))

# Load SNV profile
snv.raw0 <- readRDS("data/CRC.mut.rds")
snv.raw1 <- read.csv("./data/CRC.mut1.csv")
rownames(snv.raw1) <- colnames(snv.raw0)
colnames(snv.raw1) <- rownames(snv.raw0)
snv.raw <- as.data.frame(snv.raw1)

# Obtain shared cells
share.cell <- Reduce(intersect, list(rownames(peak.raw), rownames(cnv.raw), rownames(snv.raw)))
peak.raw <- peak.raw[share.cell,] %>% as.matrix()
cnv.raw <- cnv.raw[share.cell,] %>% as.matrix()
snv.raw <- snv.raw[share.cell,] %>% as.matrix()

# Filter peaks with low variations
peak.raw1 <- peak.raw[,apply(peak.raw, 2, sd) > 0.5]
# Adjust peaks according to copy numbers
overlapRegion <- intersectRegions(peaks = colnames(peak.raw1), cnvs = colnames(cnv.raw))
peak.corrected <- adjustPeak(peak.raw1, cnv.raw, overlapRegion, eta = 0.5)
# Raw peaks

# Calculate cell similarity
peak.similarity <- 1- as.matrix(parDist(as.matrix(peak.corrected), method = "cosine", threads = 40))
cnv.similarity <- 1- as.matrix(parDist(as.matrix(cnv.raw), method = "hamming", threads = 40))
snv.similarity <- 1- as.matrix(parDist(as.matrix(snv.raw), method = "hamming", threads = 40))
# Add similarity of a normal cell
# peak.similarity <- readRDS("peak.similarity.rds")
peak.similarity <- as.data.frame(peak.similarity)
peak.similarity$normal <- min(peak.similarity); 
peak.similarity <- rbind(peak.similarity, rep(min(peak.similarity), ncol(peak.similarity))) %>% as.matrix()

# cnv.similarity <- readRDS('cnv.similarity.rds')
norm.cnv <- rbind(cnv.raw, rep(2, ncol(cnv.raw)))
rownames(norm.cnv)[nrow(norm.cnv)] <- "normal"
cnv.similarity <- 1 - as.matrix(parDist(norm.cnv, method = "hamming", threads = 40))

# snv.similarity <- readRDS('snv.similarity.rds')
norm.snv <- rbind(snv.raw, rep(0, ncol(snv.raw)))
rownames(norm.snv)[nrow(norm.snv)] <- "normal"
snv.similarity <- 1 - as.matrix(parDist(norm.snv, method = "hamming", threads = 40))
# Running MAAS
maas_res <- MAAS(peak.similarity, cnv.similarity, snv.similarity, dims = 2:5)
saveRDS(share.cell, "maas.barcode.rds")
saveRDS(maas_res, "maas_res.rds")
write.csv(maas_res[[length(maas_res)]], "parameter.iter.csv")
